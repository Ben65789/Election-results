function parseCSV(text) { return text.split('\n').map(row => { let result = [], cur = '', inQ = false; for (let i = 0; i < row.length; i++) { let ch = row[i]; if (ch === '"') inQ = !inQ; else if (ch === ',' && !inQ) { result.push(cur.trim()); cur = ''; } else { cur += ch; } } result.push(cur.trim()); return result; }).filter(r => r.some(c => c !== '')); } async function fetchSheet(name){ let res = await fetch(`https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:csv&sheet=${name}`); let text = await res.text(); let rows = parseCSV(text); let data = { total:0, declared:0, parties:[], results:[] }; let mode = 'meta'; rows.forEach(r=>{ let key = (r[0] || '').toLowerCase(); if (key === 'total_seats') data.total = parseInt(r[1]) || 0; else if (key === 'declared') data.declared = parseInt(r[1]) || 0; else if (key === 'party') mode = 'p'; else if (key === 'area') mode = 'r'; else if (mode === 'p') { data.parties.push({ n: r[0], g: parseInt(r[1]) || 0, l: parseInt(r[2]) || 0 }); } else if (mode === 'r') { data.results.push({ a: r[0], p: r[1], m: r[2], t: r[3] }); } }); return data; }
